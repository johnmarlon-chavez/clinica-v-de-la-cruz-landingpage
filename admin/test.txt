        function initCalendar() {
            var calendarEl = document.getElementById('calendar');
            calendar = new FullCalendar.Calendar(calendarEl, {
                locale: 'es',
                initialView: 'timeGridWeek',
                headerToolbar: {
                    left: 'prev,next today',
                    center: 'title',
                    right: 'dayGridMonth,timeGridWeek,timeGridDay'
                },
                slotMinTime: "08:00:00",
                slotMaxTime: "20:00:00",
                allDaySlot: false,
                slotDuration: '00:30:00',
                businessHours: { daysOfWeek: [ 1, 2, 3, 4, 5, 6 ], startTime: '09:00', endTime: '19:00' },
                events: [
                    { title: 'Carlos Mendes - Ortodoncia', start: new Date().toISOString().split('T')[0] + 'T10:00:00', end: new Date().toISOString().split('T')[0] + 'T11:00:00', className: 'event-ortodoncia' },
                    { title: 'María Rojas - Limpieza', start: new Date().toISOString().split('T')[0] + 'T14:30:00', end: new Date().toISOString().split('T')[0] + 'T15:00:00' }
                ],
                selectable: true,
                select: function(info) {
                    const dateStr = info.startStr.split('T')[0];
                    let timeStr = info.startStr.includes('T') ? info.startStr.split('T')[1].substring(0,5) : '09:00';
                    document.getElementById('appDate').value = dateStr;
                    document.getElementById('appTime').value = timeStr;
                    openModal();
                },
                eventClick: function(info) {
                    alert('Paciente: ' + info.event.title + '\nHorario: ' + info.event.start.toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'}));
                }
            });
            calendar.render();
            
            const today = new Date().toISOString().split('T')[0];
            document.getElementById('appDate').setAttribute('min', today);
        }

        const modal = document.getElementById('appointmentModal');
        function openModal() {
            modal.classList.add('active');
            if(!document.getElementById('appDate').value) {
                document.getElementById('appDate').value = new Date().toISOString().split('T')[0];
            }
        }

        function closeModal() {
            modal.classList.remove('active');
            document.getElementById('appointmentForm').reset();
            document.getElementById('appDate').value = '';
        }

        function saveAppointment(e) {
            e.preventDefault();
            const patient = document.getElementById('patientName').value;
            const service = document.getElementById('serviceType').value;
            const date = document.getElementById('appDate').value;
            const time = document.getElementById('appTime').value;

            const startDateTime = `${date}T${time}:00`;
            const endDate = new Date(startDateTime);
            endDate.setHours(endDate.getHours() + 1);
            
            let colorCls = '';
            if(service.includes('Ortodoncia')) colorCls = 'event-ortodoncia';
            else if(service.includes('Resina') || service.includes('Curación')) colorCls = 'event-resina';
            else if(service.includes('Extracción')) colorCls = 'event-extraccion';

            calendar.addEvent({ title: `${patient} - ${service}`, start: startDateTime, end: endDate.toISOString(), className: colorCls });
            closeModal();
            showToast();
        }
        
        function showToast() {
            const toast = document.getElementById('toastMessage');
            toast.classList.add('show');
            setTimeout(() => { toast.classList.remove('show'); }, 3000);
        }
